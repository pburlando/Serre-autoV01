182 float SerreAuto::temperature_plateau() {
183 
184  int len_temptable = sizeof(temptable)/sizeof(temptable[0]);
185  float celsius = 0;
186  int i = 0;
187  int raw = analogRead(THERM0);
188 
189  while(raw > temptable[i][0]) { // c
190  // Parcourir la table pour trouver la valeur la plus proche par excés (immédiatement supérieure à raw)
191  i++;
192  if (i == len_temptable) break; // Sortir de la boucle si on a parcouru tout le tableau
193  }
194 
195  if (i == 0) {
196  celsius = temptable[i][1]; // La température est supérieure à 300°C
197  }
198 
199  else if (i == len_temptable) {
200  // La température est inférieure à 0°C
201  celsius = temptable[i - 1][1];
202  }
203 
204  else {
205  // La température est comprise entre la valeur actuelle de la table et la valeur précédente
206  // On calcule la valeur de la température par interpolation linéaire
207  celsius = temptable[i][1] + (temptable[i][0] - raw) * float((temptable[i-1][1] - temptable[i][1])) / float((temptable[i][0] - temptable[i-1][0]));
208  }
209  _temperature_plateau = celsius;
210  return _temperature_plateau;
211 }
